\item 
Find the value of \textit{k} if $\myvec{2\\1}$ is a solution of the equation\begin{align}
    \myvec{2&3}\vec{x}=\textit{k}
\end{align}
\solution
We have:\\\begin{align}
    \myvec{2&3}\vec{x}=\textit{k}\label{eq.1}
\end{align}
where,
\begin{align}
    \vec{x}=\myvec{x\\y}
\end{align}
Let \begin{align}
    \vec{A}=\myvec{2\\1}
\end{align}
be the solution of \eqref{eq.1}\\
Substituting $\vec{A}$ in \eqref{eq.1}, we get,
\begin{align}
    (2\times2)+(3\times1)&=\textit{k}
\end{align}
\begin{align}
    \therefore \textit{k}&=7
\end{align}
